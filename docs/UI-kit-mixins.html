Colors

  • LESS variable: @<color-name>
  • To use this color as background-color: add class <color-name>-bg (LESS mixin: .<color-name>-bg)
  • To use this color as text-color: add class <color-name>-txt (LESS mixin: .<color-name>-txt)

Parametric Mixins

Parametric Mixins

Parametric mixins can take arguments, which are variables passed to the block of selectors when it is mixed in.

  • .border-radius(@arguments) - generates necessary prefixes for browsers
  • .box-shadow(@arguments) - generates necessary prefixes for browsers
  • .text-shadow (@arguments) - generates necessary prefixes for browsers
  • .box-sizing (@arguments) - generates necessary prefixes for browsers
  • .border-custom-color (@border-color) - returns border: 1px solid @border-color;
  • .word-wrap (@arguments) - returns border: 1px solid @border-color;
  • .text-overflow(@arguments) - generates necessary prefixes for browsers;
  • .transition-duration (@arguments) - generates necessary prefixes for browsers
  • .gradient-vertical-bottom2top(@color, @start-color, @stop-color) - vertical gradient, from BOTTOM to TOP

.border-radius (@radius) {
  -moz-border-radius: @radius;
  -webkit-border-radius: @radius;
  border-radius: @radius;
}
.text-shadow (@arguments) {
    -moz-text-shadow: (@arguments);
    -webkit-text-shadow: (@arguments);
    text-shadow: (@arguments);
}
.box-sizing (@arguments) {
    -moz-box-sizing: (@arguments);
    -webkit-box-sizing: (@arguments);
    box-sizing: (@arguments);
}

.border-custom-color (@border-color) {
    border: 1px solid @border-color;
}

.box-shadow(@arguments) { 
  -webkit-box-shadow: @arguments;
  -moz-box-shadow: @arguments;
  box-shadow: @arguments;
}
.word-wrap (@arguments) {
    -ms-word-wrap: (@arguments);
    word-wrap: (@arguments);
}
.text-overflow(@arguments) {
    -ms-text-overflow : (@arguments);
    -o-text-overflow : (@arguments);
    text-overflow: (@arguments);
}
.transition-duration (@arguments) {
    -o-transition-duration: @arguments;
    -webkit-transition-duration: @arguments;
    -moz-transition-duration: @arguments;
    transition-duration: @arguments;
}
.transform (@arguments) {
    -webkit-transform: @arguments;
    -moz-transform:    @arguments;
    -ms-transform:     @arguments;
    -o-transform:      @arguments;
    transform:         @arguments;
}

.scale (@factor) {
	-webkit-transform: scale(@factor);
	-moz-transform: 	 scale(@factor);
	-ms-transform: 		 scale(@factor);
	-o-transform: 		 scale(@factor);
}

.animation (@name, @duration: 300ms, @delay: 0, @ease: ease) {
	-webkit-animation: @name @duration @delay @ease;
	-moz-animation:    @name @duration @delay @ease;
	-ms-animation:     @name @duration @delay @ease;
}

.rotate (@deg) {
	-webkit-transform: rotate(@deg);
	-moz-transform: 	 rotate(@deg);
	-ms-transform: 		 rotate(@deg);
	-o-transform: 		 rotate(@deg);
}
.skew (@deg, @deg2) {
	-webkit-transform:       skew(@deg, @deg2);
	-moz-transform: 	 skew(@deg, @deg2);
	-ms-transform: 		 skew(@deg, @deg2);
	-o-transform: 		 skew(@deg, @deg2);
}

.transition (@transition) {
	-webkit-transition: @transition;  
	-moz-transition: @transition;
	-ms-transition: @transition; 
	-o-transition: @transition;  
    transition: @transition;
}

.translate (@x, @y:0) {
	-webkit-transform:       translate(@x, @y);
	-moz-transform: 	 translate(@x, @y);
	-ms-transform: 		 translate(@x, @y);
	-o-transform: 		 translate(@x, @y);
}


.animation(@arguments) {
    -webkit-animation: (@arguments);
	-moz-animation: (@arguments);
	-o-animation: (@arguments);
	-ms-animation: (@arguments);
	animation: (@arguments);
}

.gradient-vertical-bottom2top(@color, @start-color, @stop-color) { 
    background: @color;
    background: -webkit-gradient(linear,left bottom,left top,color-stop(0, @start-color),color-stop(1, @stop-color));
    background: -webkit-linear-gradient(top, @stop-color, @start-color);
    background: -ms-linear-gradient(top,@start-color,@stop-color);
    background: -moz-linear-gradient(center bottom,@start-color,@stop-color);
    background: -o-linear-gradient(top, @start-color,@stop-color);
    filter: progid:DXImageTransform.Microsoft.gradient( startColorstr=@start-color, endColorstr=@stop-color,GradientType=0 );
}

.font-size (@font-size: @default-font-size) {
    @rem-size: (@font-size / @default-font-size );
    font-size: @font-size * 1px;
    font-size: ~"@{rem-size}rem"; 
}
